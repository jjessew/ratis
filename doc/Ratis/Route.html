def self.all
  response = Request.get 'Allroutes'
  return [] unless response.success?

  routes = response.to_hash[:allroutes_response][:routes].split(/\n/)
  atis_routes = routes.map do |r|
    r.strip!
    next if r.blank?
    r = r.split(/, /)
    Route.new r[0].strip, r[1..-1].map(&:strip)
  end
  atis_routes.compact
end
